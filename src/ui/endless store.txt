import tkinter as tk

class HistoryPanel:
    def __init__(self, root, callback):
        self.callback = callback

        self.label = tk.Label(root, text="Recent Calculations", font=("Arial", 14, "bold"))
        self.label.pack()

        self.listbox = tk.Listbox(root, height=10, font=("Arial", 12))
        self.listbox.pack(fill="both", padx=10, pady=10)

        self.listbox.bind("<<ListboxSelect>>", self.select_history)

    def add_item(self, item):
        self.listbox.insert(tk.END, item)

    def select_history(self, event):
        selected = self.listbox.curselection()

        if selected:
            value = self.listbox.get(selected[0])
            expression = value.split("=")[0].strip()
            self.callback(expression)
